// 기존 규칙(match /{document=**}) + 10trix 사이트별 규칙을 함께 넣은 버전
// Firebase 콘솔 > 규칙 탭에서 기존 내용 전부 지우고, 아래 전체를 붙여넣기

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // 기존 규칙 (다른 사이트용 - 그대로 유지)
    match /{document=**} {
      allow read, write: if true;
    }
    // 10trix 등 사이트별 회원/폼 (추가)
    match /sites/{siteId}/users/{userId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    match /sites/{siteId}/forms/{docId} {
      allow create: if true;
      allow read: if true;
      allow update, delete: if false;
    }
  }
}
